 86class FlagCode(enum.Enum):
 87    """Maps a flag code to a severity level.
 88
 89    Note:
 90        Severity of flag codes are as follows:
 91
 92        DEV_UNHANDLED > HALT > RED > YELLOW > GREEN > INFO > SKIPPED
 93
 94    Example:
 95        Comparing flags will compare severity levels::
 96
 97            FlagCode.RED > FlagCode.GREEN
 98            # should return True
 99
100    """
101
102    DEV_UNHANDLED = 90
103    """Never used directly, instead used when an unhandled exceptions is raised in the check function"""
104    HALT = 80
105    """Denotes an issue problematic enough to merit halting an ongoing process immediately. Note: Actually issuing any halt commands is outside the scope of validation protocol itself."""
106    RED = 50
107    """A severe issue. Likely requiring further manual investigation."""
108    YELLOW = 30
109    """A moderate/minor issue. May require further manual investigation especially if concentrated around a specific component."""
110    GREEN = 20
111    """Indicates no issues found during the check."""
112    INFO = 10
113    """Denotes the results of a check without an issue catching mechanism. May be used to report information in the flag table."""
114    SKIPPED = 1
115    """Never used directly, instead is the flag code for skipped checks"""
116    MANUAL = 2
117    """Never used directly, instead is the flag code for checks that are not automated"""
118
119    # allow comparing flag codes
120    # used to determine if a code is of higher severity
121    def __ge__(self, other):
122        return self.value >= other.value
123
124    def __le__(self, other):
125        return self.value <= other.value
126
127    def __gt__(self, other):
128        return self.value > other.value
129
130    def __lt__(self, other):
131        return self.value < other.value

157

159    and inputs to form a check queue.
160

202                    message = "Engine needs work!"

207                if cur_gas >= minimum_gas:
208                    code = FlagCode.GREEN
209                    message = f"Gas tank is at {cur_gas}. Which is above {minimum_gas}"
210                else:
211                    code = FlagCode.HALT1

213                        f"Gas tank needs a fill up! Current: {cur_gas}. Minimum: {minimum_gas}"

412    def component_start(self, name: str, description: str, skip: bool = False):
413        """Start a new component block.
414        The component will be automatically placed in the component tree.
415
416        Args:
417            name (str): Used to describe the component and for skipping by name
418            description (str): Currently unused.
419                Will be used to be a explanation of the overall goal of the component's checks
420            skip (bool, optional): Controls whether all checks under the component should be skipped.
421                This includes all child component checks. Defaults to False.
422        """
423        try:
424            # set on protocol init
425            # set current stage as parent
426            new_component = ValidationProtocol._Component(
427                name=name, parent=self.cur_component, description=description
428            )
429            self.cur_component = new_component
430
431            directly_skipped_by_protocol_init = (
432                new_component.name in self.skip_components
433            )
434            directly_skipped_by_add_call = skip
435            parent_skipping_children = new_component.parent.skip_children
436            not_in_run = not new_component.ancestry_is_in(self.run_components)
437            explicitly_run_by_protocol_init = (
438                new_component.name in self.run_components
439                and self.run_components != ValidationProtocol._ALL_COMPONENTS()
440            )
441
442            # determine if skipping needs to be set for the new component
443            if explicitly_run_by_protocol_init and not directly_skipped_by_add_call:
444                pass
445            elif any(
446                [
447                    directly_skipped_by_protocol_init,
448                    directly_skipped_by_add_call,
449                    parent_skipping_children,
450                    not_in_run,
451                ]
452            ):
453                new_component.skip = True
454                # propagate skip children
455                if any(
456                    [
457                        parent_skipping_children,
458                        directly_skipped_by_protocol_init,
459                        directly_skipped_by_add_call,
460                    ]
461                ):
462                    new_component.skip_children = True
463            yield
464        finally:
465            self.cur_component = self.cur_component.parent

413    def component_start(self, name: str, description: str, skip: bool = False):
414        """Start a new component block.
415        The component will be automatically placed in the component tree.
416
417        Args:
418            name (str): Used to describe the component and for skipping by name
419            description (str): Currently unused.
420                Will be used to be a explanation of the overall goal of the component's checks
421            skip (bool, optional): Controls whether all checks under the component should be skipped.
422                This includes all child component checks. Defaults to False.

425            # set on protocol init
426            # set current stage as parent
427            new_component = ValidationProtocol._Component(
428                name=name, parent=self.cur_component, description=description

430            self.cur_component = new_component
431
432            directly_skipped_by_protocol_init = (
433                new_component.name in self.skip_components
434            )
435            directly_skipped_by_add_call = skip
436            parent_skipping_children = new_component.parent.skip_children
437            not_in_run = not new_component.ancestry_is_in(self.run_components)
438            explicitly_run_by_protocol_init = (
439                new_component.name in self.run_components
440                and self.run_components != ValidationProtocol._ALL_COMPONENTS()
441            )
442
443            # determine if skipping needs to be set for the new component
444            if explicitly_run_by_protocol_init and not directly_skipped_by_add_call:
445                pass
446            elif any(
447                [
448                    directly_skipped_by_protocol_init,
449                    directly_skipped_by_add_call,
450                    parent_skipping_children,
451                    not_in_run,
452                ]
453            ):
454                new_component.skip = True
455                # propagate skip children
456                if any(
457                    [
458                        parent_skipping_children,
459                        directly_skipped_by_protocol_init,
460                        directly_skipped_by_add_call,
461                    ]
462                ):
463                    new_component.skip_children = True

633    def queued_checks(
634        self,
635        include_individual_checks: bool = True,
636        include_skipped_components: bool = False,
637        include_manual_checks: bool = False,
638        long_description: bool = False,
639        INDENT_CHAR: str = " ",
640        COMPONENT_PREFIX: str = "↳",
641        CHECK_PREFIX: str = " > ",
642        INDENT_CHECKS_STR: str = " ",
643        include_checks_counters: bool = True,
644        WRAP_COMPONENT_NAME_CHAR: str = "'",
645    ) -> str:
646        """Returns a print-friendly string describing the queued checks.
647
648        Args:
649            include_individual_checks (bool, optional): Controls whether individual checks should be included. Defaults to True.
650            include_skipped_components (bool, optional): Controls whether skipped checks should be included. Defaults to False.
651            long_description (bool, optional): Controls the kind of description to print. Defaults to False (i.e. use 'description' field). If set to True, the 'full_description' is used instead
652            INDENT_CHAR (str, optional): Controls the character for indenting increasing levels of component scopes. Defaults to the ' ' character.
653            COMPONENT_PREFIX (str, optional): Controls the character for to prefix all component names. Defaults to the '↳' character.
654
655        Returns:
656            str: A human friendly description of the queued checks.
657        """
658        description_field: Literal["full_description"] | Literal["description"]
659        if long_description:
660            description_field = "full_description"
661        else:
662            description_field = "description"
663
664        # create by component dictionary
665        check_by_component: dict[
666            ValidationProtocol._Component, list[ValidationProtocol._QueuedCheck]
667        ] = defaultdict(list)
668        log.info("Adding automated checks to specification")
669        for check in self._check_queue:
670            log.debug(f"Adding {check}")
671            check_by_component[check["component"]].append(check)
672        if include_manual_checks:
673            log.info("Adding manual checks to specification")
674            for check in self._manual_check_queue:
675                log.debug(f"Adding {check}")
676                check_by_component[check["component"]].append(check)
677
678        def sum_all_children(component: ValidationProtocol._Component) -> int:
679            sum = len(check_by_component[component])
680            for child in component.children:
681                sum += sum_all_children(child)
682            return sum
683
684        # print tree of components from top down
685        def render_self_and_children(component: ValidationProtocol._Component) -> str:
686            if not include_skipped_components and component.skip:
687                return ""
688            INDENT_STR = INDENT_CHAR * (len(component.ancestor_line) - 1)
689            count_str = f"[{sum_all_children(component)}"
690            count_str2 = f"[{len(check_by_component[component])}"
691            lead_str = f"{INDENT_STR}{COMPONENT_PREFIX}{WRAP_COMPONENT_NAME_CHAR}{component.name}{WRAP_COMPONENT_NAME_CHAR}{'-> !SKIPPED!' if component.skip else ''}"
692            if include_checks_counters:
693                buffer = (
694                    f"{lead_str : <55}DIRECT:{count_str2 : >4}] ALL:{count_str : >5}]"
695                )
696            else:
697                buffer = lead_str
698
699            if include_individual_checks:
700                check_lines = Counter(
701                    [
702                        f"{INDENT_CHECKS_STR}{CHECK_PREFIX}{check[description_field]}"
703                        for check in check_by_component[component]
704                        if check["to_run"]
705                    ]
706                )
707                check_line_print = [
708                    f"{line} x {line_count}" if line_count > 1 else line
709                    for line, line_count in check_lines.items()
710                ]
711                if check_lines:
712                    buffer += "\n" + "\n".join(check_line_print)
713
714            for child in component.children:
715                if render_self_and_children(child):
716                    buffer += "\n" + render_self_and_children(child)
717            return buffer
718
719        return render_self_and_children(self._root_component)

724    def run(self, flag_unhandled_exceptions: bool = False):
725        """Runs all queued checks not marked as skipped.
726
727        Args:
728            flag_unhandled_exceptions (bool, optional): If true, any unhandled exceptions within
729                check functions will be flagged with a DEV_UNHANDLED flag but
730                will not halt the generation of the overall protocol report.
731                Defaults to False which means the protocol will raise the exception and stop midway.
732        """
733        all_queued = self._check_queue + self._manual_check_queue
734        for queued in all_queued:
735            fcn = queued["check_fcn"]
736            if queued["automated"]:
737                fcn_name = fcn.__name__
738            else:
739                fcn_name = "MANUAL_CHECK"
740
741            if not queued["to_run"]:
742                # case: skipping compute
743                packed_result = {
744                    "code": FlagCode.SKIPPED,
745                    "message": "Skipped by protocol",
746                    "function": fcn_name,
747                    "description": queued["description"],
748                    # NOTE: This is intentionally the unevaluated payload
749                    #     This assumes skipped checks may contain inputs that fail on evaluation
750                    #     One example being data models that aren't fully loaded for skipped checks
751                    "kwargs": queued["payload"] | queued["config"],
752                    "config": queued["config"],
753                }
754
755            elif not queued["automated"]:
756                # case: requires human analyst
757                packed_result = {
758                    "code": FlagCode.MANUAL,
759                    "message": "Not automated: Pending analyst review",
760                    "function": fcn_name,
761                    "description": queued["description"],
762                    # NOTE: This is intentionally the unevaluated payload
763                    #     This assumes skipped checks may contain inputs that fail on evaluation
764                    #     One example being data models that aren't fully loaded for skipped checks
765                    "kwargs": queued["payload"] | queued["config"],
766                    "config": queued["config"],
767                }
768
769            else:
770                # case: to_compute is true
771                # try computing, works
772                try:
773                    payload = self._eval_payload_callables(queued["payload"])
774                except Exception as e:
775                    raise RuntimeError(
776                        f"Failed to evaluate payload: component:{queued['component']}, "
777                        f"check function:{fcn_name}"
778                    ) from e
779                payload_and_config = payload | queued["config"]
780                try:
781                    result = fcn(**payload_and_config)
782
783                    # peel off outlier data and keep track
784                    # using current component name as the top level key
785                    # Type ignored since FlagEntry dicts will return None as desired for this conditional
786                    if fcn_outliers := result.pop("outliers", None):  # type: ignore
787                        self.outliers[queued["component"].name] = (
788                            self.outliers[queued["component"].name] | fcn_outliers
789                        )
790
791                    # note: we want to make sure to override description if the result gives one;
792                    # however it is optional and the fallback is the function name
793                    packed_result = (
794                        {"description": queued["description"]}
795                        | result
796                        | {
797                            "kwargs": payload_and_config,
798                            "config": queued["config"],
799                            "function": fcn_name,
800                        }
801                    )
802                # except: computing failed
803                # on exception, capture exception
804                except Exception as e:
805                    if flag_unhandled_exceptions:
806                        packed_result = {
807                            "code": FlagCode.DEV_UNHANDLED,
808                            "message": str(e),
809                            "function": fcn_name,
810                            "description": queued["description"],
811                            "kwargs": payload_and_config,
812                            "config": queued["config"],
813                        }
814                    else:
815                        raise RuntimeError(
816                            f"Function failed: {fcn_name} part of {queued['component']} with original error message: {e}"
817                        ) from e
818
819            # add result (including skip flag) to component
820            queued["component"].flags.append(packed_result)

A validation protocol.

Validation protocols are used to connect check function, function configuration and inputs to form a check queue.

The check queue can then be printed and/or run.

Finally, the protocol object includes reporting functions to present the results of those checks.
