 17def isa_investigation_subtables(ISAarchive: Path) -> dict[str, pd.DataFrame]:
 18    tables: dict[str, pd.DataFrame] = dict()
 19
 20    # track sub table lines
 21    table_lines: List[list] = list()
 22    key: str = None  # type: ignore
 23
 24    try:
 25        [i_file] = (
 26            f
 27            for f in BulkRNASeqMetadataComponent.fetch_isa_files_external(ISAarchive)
 28            if f.name.startswith("i_")
 29        )
 30    except ValueError:
 31        raise FileNotFoundError(
 32            f"Could not find an i_* file inside: {ISAarchive.name}, is this an ISA archive?"
 33        )
 34    with open(i_file, "r") as f:
 35        for line in [l.rstrip() for l in f.readlines()]:
 36            # search for header
 37            if line in BulkRNASeqMetadataComponent._ISA_INVESTIGATION_HEADERS:
 38                if key != None:
 39                    tables[key] = pd.DataFrame(
 40                        table_lines
 41                    ).T  # each subtable is transposed in the i_file
 42                    table_lines = list()
 43                key = line  # set next table key
 44            else:
 45                tokens = line.split("\t")  # tab separated
 46                table_lines.append(tokens)
 47    tables[key] = pd.DataFrame(
 48        table_lines
 49    ).T  # each subtable is transposed in the i_file
 50
 51    # reformat each table
 52    def clean_quotes(string: str) -> str:
 53        SINGLE_OR_DOUBLE_QUOTES = "\"'"
 54        # don't perform on non-string elements
 55        if not isinstance(string, str):
 56            return string
 57        else:
 58            return string.lstrip(SINGLE_OR_DOUBLE_QUOTES).rstrip(
 59                SINGLE_OR_DOUBLE_QUOTES
 60            )
 61
 62    df: pd.DataFrame
 63    for key, df in tables.items():
 64
 65        # note: as a ref, no reassign needed
 66        tables[key] = (
 67            df.rename(columns=df.iloc[0]).drop(df.index[0]).applymap(clean_quotes)
 68        )
 69
 70    # ensure all expected subtables present
 71    assert set(tables.keys()) == BulkRNASeqMetadataComponent._ISA_INVESTIGATION_HEADERS
 72
 73    return tables

 76def get_assay_table_path(
 77    ISAarchive: Path, configuration: dict, return_index: bool = False
 78) -> Path:
 79    """Retrieve the assay table file name that determined as a valid assay based on configuration.
 80    Specifically, defined in subsection 'ISA meta'
 81
 82    :param study_assay_table: From the investigation file
 83    :type study_assay_table: pd.DataFrame
 84    :param configuration: Standard assay parsed config
 85    :type configuration: dict
 86    :return: Path to the found assay table
 87    :rtype: Path
 88    """
 89    config = configuration["ISA Meta"]
 90    # retrieve study assay subtable from I_file
 91    df = isa_investigation_subtables(ISAarchive)["STUDY ASSAYS"]
 92
 93    # get valid tuples of measurement and technology types from configuration
 94    valid_measurements_and_technology_types: list[tuple[str, str]] = [
 95        (entry["measurement"], entry["technology"])
 96        for entry in config["Valid Study Assay Technology And Measurement Types"]
 97    ]
 98
 99    # check for matching rows based on configuration tuple
100    # one and only one row should match
101    # not very efficient, but table should never be too large for this to be of concern
102    matches: list[Path] = list()
103    for valid_combination in valid_measurements_and_technology_types:
104        log.debug(f"Searching subtable for {valid_combination}")
105        match_row = df.loc[
106            (
107                df[["Study Assay Measurement Type", "Study Assay Technology Type"]]
108                == valid_combination
109            ).all(axis="columns")
110        ]
111        match_file = [Path(val) for val in match_row["Study Assay File Name"].values]
112        matches.extend(match_file)
113
114    # guard, one and only one should match
115    assert (
116        len(matches) == 1
117    ), f"One and only one should match, instead got these matches: {matches} using there queries: {valid_measurements_and_technology_types} against these listed in the ISA archive: {df[['Study Assay Measurement Type', 'Study Assay Technology Type']]}"
118
119    # load assay table
120    assay_file_path = matches[0]
121    [assay_path] = [
122        f
123        for f in BulkRNASeqMetadataComponent.fetch_isa_files_external(ISAarchive)
124        if f.name == assay_file_path.name
125    ]
126
127    if return_index:
128        for valid_combination in valid_measurements_and_technology_types:
129            [match_index] = df.loc[
130                (
131                    df[["Study Assay Measurement Type", "Study Assay Technology Type"]]
132                    == valid_combination
133                ).all(axis="columns")
134            ].index.values
135            return match_index
136
137    return assay_path

178def get_column_name(df: pd.DataFrame, target: Union[str, list]) -> str:
179    try:
180        match target:
181            case str():
182                [target_col] = (col for col in df.columns if col in target)
183                return target_col
184            case list():
185                for query in target:
186                    try:
187                        [target_col] = (col for col in df.columns if col in query)
188                        return target_col
189                    except ValueError:
190                        continue
191                # if this runs, the list did not match anything!
192                raise ValueError(
193                    f"Could not find required column '{target}' "
194                    f"in either ISA sample or assay table. These columns were found: {list(df.columns)}"
195                )
196    except ValueError as e:
197        raise ValueError(
198            f"Could not find required column '{target}' "
199            f"in either ISA sample or assay table. These columns were found: {list(df.columns)}"
200        ) from e

204def isa_to_runsheet(accession: str, isa_archive: Path, config: tuple[str, str]):
205    ################################################################
206    ################################################################
207    # SETUP CONFIG AND INPUT TABLES
208    ################################################################
209    ################################################################
210    log.info("Setting up to generate runsheet dataframe")
211    configuration = load_config(config=config)
212    runsheet_schema = schemas.runsheet[config[0]]
213    i_tables = isa_investigation_subtables(isa_archive)
214    a_table = pd.read_csv(
215        get_assay_table_path(ISAarchive=isa_archive, configuration=configuration),
216        sep="\t",
217    )
218    a_study_assays_index = get_assay_table_path(
219        ISAarchive=isa_archive, configuration=configuration, return_index=True
220    )
221    [s_file] = (
222        f
223        for f in BulkRNASeqMetadataComponent.fetch_isa_files_external(isa_archive)
224        if f.name.startswith("s_")
225    )
226    s_table = pd.read_csv(s_file, sep="\t")
227    df_merged = s_table.merge(a_table, on="Sample Name").set_index(
228        "Sample Name", drop=True
229    )
230
231    ################################################################
232    ################################################################
233    # GENERATE FINAL DATAFRAME
234    ################################################################
235    ################################################################
236    log.info("Generating runsheet dataframe")
237    df_final = pd.DataFrame(index=df_merged.index)
238    # extract from Investigation table first
239    investigation_source_entries = [
240        entry
241        for entry in configuration["Staging"]["General"]["Required Metadata"][
242            "From ISA"
243        ]
244        if entry["ISA Table Source"] == "Investigation"
245    ]
246    for entry in investigation_source_entries:
247        # handle special cases
248        if entry.get("True If Includes At Least One"):
249            overlap = set(entry["True If Includes At Least One"]).intersection(
250                set(i_tables[entry["Investigation Subtable"]][entry["ISA Field Name"]])
251            )
252            df_final[entry["Runsheet Column Name"]] = bool(overlap)
253            continue
254
255        target_investigation_column = i_tables[entry["Investigation Subtable"]].loc[
256            a_study_assays_index
257        ]
258        df_final[entry["Runsheet Column Name"]] = target_investigation_column[
259            entry["ISA Field Name"]
260        ]
261
262    # extract from assay table first
263    assay_source_entries = [
264        entry
265        for entry in configuration["Staging"]["General"]["Required Metadata"][
266            "From ISA"
267        ]
268        if entry["ISA Table Source"] in ["Assay", "Sample", ["Assay", "Sample"]]
269        and entry.get("Autoload", True) != False
270    ]
271    for entry in assay_source_entries:
272        assert list(df_final.index) == list(df_merged.index)
273        if entry.get("Runsheet Index"):
274            # already set and checked above
275            continue
276        else:
277            # merged sequence data file style extraction
278            if entry.get("Multiple Values Per Entry"):
279                # getting compatible column
280                target_col = get_column_name(df_merged, entry["ISA Field Name"])
281
282                # split into separate values
283                values: pd.DataFrame = df_merged[target_col].str.split(
284                    pat=entry["Multiple Values Delimiter"], expand=True
285                )
286
287                # rename columns with runsheet names, checking if optional columns are included
288                runsheet_col: dict
289                for runsheet_col in entry["Runsheet Column Name"]:
290                    if runsheet_col["index"] in values.columns:
291                        values = values.rename(
292                            columns={runsheet_col["index"]: runsheet_col["name"]}
293                        )
294                    else:  # raise exception if not marked as optional
295                        if not runsheet_col["optional"]:
296                            raise ValueError(
297                                f"Could not populate runsheet column (config: {runsheet_col}). Data may be missing in ISA or the configuration may be incorrect"
298                            )
299
300                if entry.get("GLDS URL Mapping"):
301                    urls = get_urls(accession=accession)
302
303                    def map_url_to_filename(fn: str) -> str:
304                        try:
305                            return urls.get(fn, dict())["url"]
306                        except KeyError:
307                            raise ValueError(
308                                f"{fn} does not have an associated url in {urls}"
309                            )
310
311                    values2 = values.applymap(
312                        map_url_to_filename
313                    )  # inplace operation doesn't seem to work
314                else:
315                    values2 = values
316
317                # add to final dataframe and check move onto entry
318                df_final = df_final.join(values2)
319                continue
320
321            # factor value style extraction
322            if entry.get("Matches Multiple Columns") and entry.get("Match Regex"):
323                # find matching columns
324                match_cols = [
325                    (i, col, df_merged[col])
326                    for i, col in enumerate(df_merged.columns)
327                    if re.match(pattern=entry.get("Match Regex"), string=col)
328                ]
329
330                # check if columns require appending unit
331                if entry.get("Append Column Following"):
332                    match_i: int  # index in matching column list
333                    df_i: int  # index in merged dataframe
334                    col: str
335                    original_series: pd.Series
336                    for match_i, (df_i, col, original_series) in enumerate(match_cols):
337                        # scan through following columns
338                        for scan_col in df_merged.iloc[:, df_i:].columns:
339                            # check if another 'owner' column is scanned, this means Unit was not found
340                            if any(
341                                [
342                                    scan_col.startswith("Parameter Value["),
343                                    scan_col.startswith("Factor Value["),
344                                    scan_col.startswith("Characteristics ["),
345                                ]
346                            ):
347                                break
348                            if scan_col == entry.get("Append Column Following"):
349                                resolved_series = original_series + df_merged[scan_col]
350                                match_cols[match_i] = df_i, col, resolved_series
351                                break
352
353                # finally add this information into dataframe
354                for _, col_name, series in match_cols:
355                    df_final[col_name] = series
356            else:
357                # CAUTION: normally this wouldn't be safe as the order of rows isn't considered.
358                # In this block, the indices are checked for parity already making this okay
359                if entry.get("Value If Not Found"):
360                    try:
361                        target_col = get_column_name(df_merged, entry["ISA Field Name"])
362                        series_to_add = df_merged[target_col]
363                    except ValueError:
364                        series_to_add = pd.DataFrame(
365                            data={
366                                "FALLING_BACK_TO_DEFAULT": entry.get(
367                                    "Value If Not Found"
368                                )
369                            },
370                            index=df_merged.index,
371                        )
372                else:
373                    target_col = get_column_name(df_merged, entry["ISA Field Name"])
374                    series_to_add = df_merged[target_col]
375                if entry.get("GLDS URL Mapping"):
376                    urls = get_urls(accession=accession)
377
378                    def map_url_to_filename(fn: str) -> str:
379                        try:
380                            return urls.get(fn, dict())["url"]
381                        except KeyError:
382                            raise ValueError(
383                                f"{fn} does not have an associated url in {urls}"
384                            )
385
386                    _swap = series_to_add.map(
387                        map_url_to_filename
388                    )  # inplace operation doesn't seem to work
389                    series_to_add = _swap
390                if entry.get("Remapping"):
391                    df_final[entry["Runsheet Column Name"]] = series_to_add.map(
392                        lambda val: entry.get("Remapping")[val]
393                    )
394                else:
395                    df_final[entry["Runsheet Column Name"]] = series_to_add
396    ################################################################
397    ################################################################
398    # PREPROCESSING
399    # - Create new column
400    #   - Original Sample Name (used for post processing consistency)
401    # - Reworks Sample Name for processing compatibility
402    ################################################################
403    ################################################################
404    # to preserve the original sample name for post processing
405    # make a new column
406    df_final["Original Sample Name"] = df_final.index
407
408    # then modify the index as needed
409    df_final.index = df_final.index.str.replace(" ", "_")
410    modified_samples: list[str] = list(
411        df_final.loc[df_final.index != df_final["Original Sample Name"]].index
412    )
413    if len(modified_samples) != 0:
414        log.info(
415            f"The following orignal sample names modified for processing: {modified_samples}"
416        )
417
418    ################################################################
419    ################################################################
420    # VALIDATION
421    ################################################################
422    ################################################################
423    # TODO: Need to make the validation generalized, maybe load a validation object based on a configuration key?
424    log.info("Validating runsheet dataframe")
425    # validate dataframe contents (incomplete but catches most required columns)
426    # uses dataframe to dict index format: https://pandas.pydata.org/docs/reference/api/pandas.DataFrame.to_dict.html
427
428    runsheet_schema.validate(df_final.to_dict(orient="index"))
429    # ensure at least on Factor Value is extracted
430    assert (
431        len([col for col in df_final.columns if col.startswith("Factor Value[")]) != 0
432    ), f"Must extract at least one factor value column but only has the following columns: {df_final.columns}"
433
434    ################################################################
435    ################################################################
436    # WRITE OUTPUT
437    ################################################################
438    ################################################################
439    # output file
440    output_fn = (
441        f"{accession}_{configuration['NAME']}_v{configuration['VERSION']}_runsheet.csv"
442    )
443    log.info(
444        f"Writing runsheet to: {output_fn} with {df_final.shape[0]} rows and {df_final.shape[1]} columns"
445    )
446    df_final.to_csv(output_fn)
447
448    return df_final

240    investigation_source_entries = [
241        entry
242        for entry in configuration["Staging"]["General"]["Required Metadata"][
243            "From ISA"
244        ]
245        if entry["ISA Table Source"] == "Investigation"
246    ]
247    for entry in investigation_source_entries:
248        # handle special cases
249        if entry.get("True If Includes At Least One"):
250            overlap = set(entry["True If Includes At Least One"]).intersection(
251                set(i_tables[entry["Investigation Subtable"]][entry["ISA Field Name"]])
252            )
253            df_final[entry["Runsheet Column Name"]] = bool(overlap)
254            continue
255
256        target_investigation_column = i_tables[entry["Investigation Subtable"]].loc[
257            a_study_assays_index
258        ]
259        df_final[entry["Runsheet Column Name"]] = target_investigation_column[
260            entry["ISA Field Name"]
261        ]
262
263    # extract from assay table first
264    assay_source_entries = [
265        entry
266        for entry in configuration["Staging"]["General"]["Required Metadata"][
267            "From ISA"
268        ]
269        if entry["ISA Table Source"] in ["Assay", "Sample", ["Assay", "Sample"]]
270        and entry.get("Autoload", True) != False
271    ]
272    for entry in assay_source_entries:
273        assert list(df_final.index) == list(df_merged.index)
274        if entry.get("Runsheet Index"):
275            # already set and checked above
276            continue
277        else:
278            # merged sequence data file style extraction
279            if entry.get("Multiple Values Per Entry"):
280                # getting compatible column
281                target_col = get_column_name(df_merged, entry["ISA Field Name"])
282
283                # split into separate values
284                values: pd.DataFrame = df_merged[target_col].str.split(
285                    pat=entry["Multiple Values Delimiter"], expand=True
286                )
287
288                # rename columns with runsheet names, checking if optional columns are included
289                runsheet_col: dict
290                for runsheet_col in entry["Runsheet Column Name"]:
291                    if runsheet_col["index"] in values.columns:
292                        values = values.rename(
293                            columns={runsheet_col["index"]: runsheet_col["name"]}
294                        )
295                    else:  # raise exception if not marked as optional
296                        if not runsheet_col["optional"]:
297                            raise ValueError(
298                                f"Could not populate runsheet column (config: {runsheet_col}). Data may be missing in ISA or the configuration may be incorrect"
299                            )
300
301                if entry.get("GLDS URL Mapping"):
302                    urls = get_urls(accession=accession)
303
304                    def map_url_to_filename(fn: str) -> str:
305                        try:
306                            return urls.get(fn, dict())["url"]
307                        except KeyError:
308                            raise ValueError(
309                                f"{fn} does not have an associated url in {urls}"
310                            )
311
312                    values2 = values.applymap(
313                        map_url_to_filename
314                    )  # inplace operation doesn't seem to work
315                else:
316                    values2 = values
317
318                # add to final dataframe and check move onto entry
319                df_final = df_final.join(values2)
320                continue
321
322            # factor value style extraction
323            if entry.get("Matches Multiple Columns") and entry.get("Match Regex"):
324                # find matching columns
325                match_cols = [
326                    (i, col, df_merged[col])
327                    for i, col in enumerate(df_merged.columns)
328                    if re.match(pattern=entry.get("Match Regex"), string=col)
329                ]
330
331                # check if columns require appending unit
332                if entry.get("Append Column Following"):
333                    match_i: int  # index in matching column list
334                    df_i: int  # index in merged dataframe
335                    col: str
336                    original_series: pd.Series
337                    for match_i, (df_i, col, original_series) in enumerate(match_cols):
338                        # scan through following columns
339                        for scan_col in df_merged.iloc[:, df_i:].columns:
340                            # check if another 'owner' column is scanned, this means Unit was not found
341                            if any(
342                                [
343                                    scan_col.startswith("Parameter Value["),
344                                    scan_col.startswith("Factor Value["),
345                                    scan_col.startswith("Characteristics ["),
346                                ]
347                            ):
348                                break
349                            if scan_col == entry.get("Append Column Following"):
350                                resolved_series = original_series + df_merged[scan_col]
351                                match_cols[match_i] = df_i, col, resolved_series
352                                break
353
354                # finally add this information into dataframe
355                for _, col_name, series in match_cols:
356                    df_final[col_name] = series
357            else:
358                # CAUTION: normally this wouldn't be safe as the order of rows isn't considered.
359                # In this block, the indices are checked for parity already making this okay
360                if entry.get("Value If Not Found"):
361                    try:
362                        target_col = get_column_name(df_merged, entry["ISA Field Name"])
363                        series_to_add = df_merged[target_col]
364                    except ValueError:
365                        series_to_add = pd.DataFrame(
366                            data={
367                                "FALLING_BACK_TO_DEFAULT": entry.get(
368                                    "Value If Not Found"
369                                )
370                            },
371                            index=df_merged.index,
372                        )
373                else:
374                    target_col = get_column_name(df_merged, entry["ISA Field Name"])
375                    series_to_add = df_merged[target_col]
376                if entry.get("GLDS URL Mapping"):
377                    urls = get_urls(accession=accession)
378
379                    def map_url_to_filename(fn: str) -> str:
380                        try:
381                            return urls.get(fn, dict())["url"]
382                        except KeyError:
383                            raise ValueError(
384                                f"{fn} does not have an associated url in {urls}"
385                            )
386
387                    _swap = series_to_add.map(
388                        map_url_to_filename
389                    )  # inplace operation doesn't seem to work
390                    series_to_add = _swap
391                if entry.get("Remapping"):
392                    df_final[entry["Runsheet Column Name"]] = series_to_add.map(
393                        lambda val: entry.get("Remapping")[val]
394                    )
395                else:
396                    df_final[entry["Runsheet Column Name"]] = series_to_add
397    ################################################################
398    ################################################################
399    # PREPROCESSING
400    # - Create new column
401    #   - Original Sample Name (used for post processing consistency)
402    # - Reworks Sample Name for processing compatibility
403    ################################################################
404    ################################################################
405    # to preserve the original sample name for post processing
406    # make a new column
407    df_final["Original Sample Name"] = df_final.index
408
409    # then modify the index as needed
410    df_final.index = df_final.index.str.replace(" ", "_")
411    modified_samples: list[str] = list(
412        df_final.loc[df_final.index != df_final["Original Sample Name"]].index
413    )
414    if len(modified_samples) != 0:
415        log.info(
416            f"The following orignal sample names modified for processing: {modified_samples}"
417        )
418
419    ################################################################
420    ################################################################
421    # VALIDATION
422    ################################################################
423    ################################################################
424    # TODO: Need to make the validation generalized, maybe load a validation object based on a configuration key?
425    log.info("Validating runsheet dataframe")
426    # validate dataframe contents (incomplete but catches most required columns)
427    # uses dataframe to dict index format: https://pandas.pydata.org/docs/reference/api/pandas.DataFrame.to_dict.html
428
429    runsheet_schema.validate(df_final.to_dict(orient="index"))
430    # ensure at least on Factor Value is extracted
431    assert (
432        len([col for col in df_final.columns if col.startswith("Factor Value[")]) != 0
433    ), f"Must extract at least one factor value column but only has the following columns: {df_final.columns}"
434
435    ################################################################
436    ################################################################
437    # WRITE OUTPUT
438    ################################################################
439    ################################################################
440    # output file
441    output_fn = (
442        f"{accession}_{configuration['NAME']}_v{configuration['VERSION']}_runsheet.csv"
443    )
444    log.info(
445        f"Writing runsheet to: {output_fn} with {df_final.shape[0]} rows and {df_final.shape[1]} columns"
446    )
447    df_final.to_csv(output_fn)
448
449    return df_final
